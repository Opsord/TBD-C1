Create table Productos(
	idProducto Integer Primary Key,
	nombreProducto varchar(255),
	descripcionProducto varchar(255),
	precioProducto Integer,
	stockProducto Integer
	
);



Create table Venta(
	idVenta Integer Primary Key,
	fechaVenta Date,
	montoVenta Integer
);



Create table Tienda(
	idTienda Integer Primary Key,
	nombreTienda varchar(255),
	direccionTienda varchar(255),
        comunaTienda varchar(255),
        Foreign Key (comunaTienda) References Comuna(idComuna),
);



Create Table TipoDocumento(
	idTipoDocumento Integer Primary Key,
	nombreDocumento varchar(255)
);



Create Table Vendedor (
	idVendedor Integer Primary Key,
	comisionVendedor Integer,
	horasExtras Integer
);



Create Table Comuna (
	idComuna Integer Primary Key,
	nombreComuna varchar(255),
	prvinciaComuna varchar(255),
	regionComuna varchar(255)
);



Create Table Sueldo (
	idSueldo Integer Primary Key,
	montoSueldo Integer,
	fechaPago Date
);



Create Table Empleado(
	idEmpleado Integer Primary key,
	nombreEmpleado varchar(255),
	apellidoEmpleado varchar(255),
	telefonoEmpleado varchar(255),
	correoEmpleado varchar(255),
	fechaContrato Date
        comunaEmpleado varchar(255),
        Foreign Key (comunaEmpleado) References Comuna(idComuna),

);


Create Table Producto_Venta (
	idProductoVenta Integer Primary Key,
	idProd Integer,
	idVent Integer,
	cantidadVendida integer,
	Foreign Key (idProd) References Productos(idProducto),
	Foreign Key (idVent) References Venta(idVenta)
	
);



Create Table Venta_Vendedor (
	idVentaVendedor Integer Primary Key,
	idVend Integer,
	idEmpl Integer,
	Foreign Key (idVend) References Vendedor(idVendedor),
	Foreign Key (idEmpl) References Empleado(idEmpleado)
);




Create Table Tienda_Empleado(
	idTiendaEmpleado Integer Primary Key,
	idEmplTienda Integer,
	idTiendaEmpl Integer,
	Foreign Key (idEmplTienda) References Empleado(idEmpleado),
	Foreign Key (idTiendaEmpl) References Tienda(idTienda)
);
