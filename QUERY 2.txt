QUERY 2

SELECT t.idTienda,
       t.NombreTienda,
       p.IdProducto,
       p.NombreProducto,
       p.PrecioProducto AS PrecioMasEconomico
FROM Tienda t
JOIN Producto_Tienda pt ON t.idTienda = pt.IdTienda
JOIN Productos p ON pt.IdProducto = p.IdProducto
JOIN (SELECT IdTienda, MIN(PrecioProducto) AS PrecioMinimo
      FROM Producto_Tienda pt
      JOIN Productos p ON pt.IdProducto = p.IdProducto
      GROUP BY IdTienda) AS Subconsulta
ON pt.IdTienda = Subconsulta.IdTienda AND p.PrecioProducto = Subconsulta.PrecioMinimo;
